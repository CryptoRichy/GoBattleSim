// Manually Modify Data
function manuallyModifyData(data){
	if (data.FastMoves){
		
		var fmove_transform = getEntry('transform', data.FastMoves);
		if (fmove_transform){
			fmove_transform.effect_name = "transform";
		}
		var hidden_power_variations = [];
		for (var type in Data.TypeEffectiveness){
			insertEntry({
			  "name": "hidden power " + type,
			  "power": 15,
			  "pokeType": type,
			  "dws": 1100,
			  "duration": 1500,
			  "label": "Hidden Power " + toTitleCase(type),
			  "icon": "https://pokemongo.gamepress.gg/sites/pokemongo/files/icon_" + type + ".png",
			  "moveType": "fast",
			  "energyDelta": 15
			}, hidden_power_variations);
		}
		data.FastMoves = mergeDatabase(hidden_power_variations, data.FastMoves);
	}
	
	if (data.Pokemon){
		var modifiedSpecies = [];
		modifiedSpecies = mergeDatabase(CASTFORM_FORMS, modifiedSpecies);
		data.Pokemon = mergeDatabase(modifiedSpecies, data.Pokemon);
		
	}


}
